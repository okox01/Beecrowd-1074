#include <stdio.h>

int main(){
    
int n,num[10000];
    
scanf("%d",&n);
    
for(int i=0;i<n;i++){
        
scanf("%d",&num[i]);
    
}
    
for(int i=0;i<n;i++){
        
if(num[i]>0){

            
if(num[i]%2==0){
                
printf("EVEN POSITIVE\n");
            
}else{
                
printf("EVEN NEGATIVE\n");
            
}
        
}
        
else if(num[i]<0){
            
if(num[i]%2==1){
                
printf("ODD POSITIVE\n");
            
}else{
            
printf("ODD NEGATIVE\n");
            
}
        
}
        
else{
            
printf("NULL\n");
        
}
    
}
    
return 0;
}
